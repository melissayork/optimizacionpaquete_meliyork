def ackley(x):
    a = 20
    b = 0.2
    c = 2 * np.pi
    suma1 = x[0]**2 + x[1]**2
    suma2 = np.cos(c * x[0]) + np.cos(c * x[1])
    term1 = -a * np.exp(-b * np.sqrt(0.5 * suma1))
    term2 = -np.exp(0.5 * suma2)
    resul = term1 + term2 + a + np.exp(1)
    return resul

def shekel(x):
    a = np.array([[4, 4, 4, 4],
                [1, 1, 1, 1]])
    c = np.array([0.1, 0.2, 0.2, 0.4])
    result = 0
    for i in range(len(c)):
        result += 1 / (c[i] + (x[0] - a[0, i])**2 + (x[1] - a[1, i])**2)
    return result
